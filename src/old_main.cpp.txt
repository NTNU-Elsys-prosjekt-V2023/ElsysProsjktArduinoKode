void songLights(int BPM){
  //A function that light up the leds that should tell the user where it is in the song
  //We first need to ger what kind of BMP we are using(what mood)
  //A loop that lights up one and one light for each part of the song
  if (changeSection)
  {
      currentLight = -1; 
      changeSection = false;
  }
  int interval = (BPM/60)*4*1000;
  //The first light is the 143 light
  int startSectionLight = section * LightsPerSector +143; 

  //TURN ON PREVIOUS LIGHTS
  if (currentLight <= startSectionLight) {
      currentLight = startSectionLight;
      //Lights up first led in MixMasters section
      leds[currentLight]=CRGB::Green;
      //we are not in the intro lets light up the leds before 
      if (currentLight != 143)
      {
        for (int i = 143; i < currentLight; i++)
        {
          leds[i]=CRGB::Green;
        }
        
      }
      
    }
  //Turn of all lights before selected section
  if (prevStopState == true)
  {
    Serial.print("Turn off all prevleds\n");
    for (int i = 143; i < 143+3*LightsPerSector; i++)
    {
      leds[i]=CRGB::Black; 
    }
    FastLED.show();
    prevStopState = false;
  }
  
  //This lights up the next light after it has gone some time
  //checking if the system is playing sound and that the mood MixBox is added
  //Does only light up leds if the system is playing a sound and it has not happened a recent change in the melody, and if the stop button has not been pressed
  if (systemPlayingSound == 1 && BPM != 0 && pauseState == 1 && changeHasCome == 0)
  {
    //only if the system is playing sound are we going to update currentMillis 
    currentMillis = millis();
    //Lighting up the first led in the right section 
    //After the beat has gone through 4 beats we are going to light up the next light
    if (currentMillis - previousMillis >= interval)
    {
      previousMillis = currentMillis;
      currentLight += 1;
      // Serial.print("Her er current light\n");
      // Serial.print(currentLight);
      //NEED TO ADD THAT THE LOOP IS 0 LIKE STOP VALUE
      if (currentLight >= LightsPerSector*(section+1) + 143 && loopState == 0)
      {
       //We have know comed too the last led and need to restart the loop
       for (int i = 143; i <LightsPerSector*section+143; i++)
       {
        //turning of all the lights
      //Maybe the board should give the user some way to know they are in a loop?????????
        leds[i]=CRGB::Black;
       }
       currentLight = startSectionLight;
       //Setting currentLight so that the loop can repeat itself
      }
      Serial.print("Light up next light\n");
      leds[currentLight]=CRGB::Green;
    }
    if (currentLight == 143+24)
    {
      Serial.print("TurnOff lights \n");
      //set current light to start light
      currentLight = 143;
      for (int i = 143;i < 143+24; i++)
      {
        leds[i]=CRGB::Black;
      }
    }
    
  }
  else if (pauseState == 0)
  {
    Serial.print("Stop leds\n");
    for (int i = 143; i < (LightsPerSector*3+143); i++)
    {
      //Making the leds glow red if the button is pressed
      leds[i]=CRGB::Black;
    }
    for (int i = 143; i < (LightsPerSector*3+143); i++)
    {
      leds[i]=CRGB::Red;
    }
    currentLight = section * LightsPerSector +143; 
    
  }
  else if (changeHasCome == 1)
  {
    Serial.print("Change has come turn off lights \n");
    for (int i = 143; i < LightsPerSector*3+143; i++)
    {
      //turning all of the leds off and the light up section will repeat it self in the next loop
      leds[i]=CRGB::Black;
    }  
    currentLight = section * LightsPerSector +143; 
  }
  FastLED.show();
}



void progressBar(int BPM){
  //pause mode light up red

  currentProgressTime = millis();
  float MaxTimeLed = ((32.0 * 3.0) / (BPM))*60000;

  
  if (!pauseState)
  {
    for (int i = 143; i < (LightsPerSector*3+143); i++)
    {
      leds[i]=CRGB::Red;
    }
    currentLight = section * LightsPerSector +143; 
    
  }
  else{

      if (currentProgressTime > progressBarStart + MaxTimeLed)
      {
        progressBarStart = currentProgressTime;
        Serial.println("progress time restarted");
      }

    int color_list[] = {180,180,180,180, 180,180,180,180,
                        200,200,200,200, 200,200,200,200,
                        220,220,220,220, 220,220,220,220};
    float totalBrightness = 8.0*3.0;
    unsigned long my_time = (currentProgressTime-progressBarStart);
    float currentBrightness = (currentProgressTime-progressBarStart)/MaxTimeLed*totalBrightness;
    for (int i = 143; i < (LightsPerSector*3+143); i++)
    {
      if (currentBrightness >= 1)
      {
        
        leds[i] = CHSV(color_list[i-143],200,255);
        currentBrightness-=1;
      }
      else{
        leds[i] = CHSV(color_list[i-143],200, currentBrightness * 255);
        currentBrightness = 0;
      }
    }
  }
  FastLED.show();
}

    //Serial.print("Setter i gang readFunksjon");

void sendMidi(){
  //Here Midisignals are being sent
  //Getting each 6-bit word from each section ready to be sent on the right channel 
  if (!sendMeIntro1)
  {
    sendMeIntroOrdA = sendMePreviousIntroOrdA;
  }
  if (!sendMeIntro2)
  {
    sendMeIntroOrdB = sendMePreviousIntroOrdB;
  }
  if (!sendMeVers1)
  {
    sendMeVersOrdA = sendMePreviousVersOrdA;
  }
  if (!sendMeVers2)
  {
    sendMeVersOrdB = sendMePreviousVersOrdB;
  }
  if (!sendMeRefreng1)
  {
    sendMeRefrengOrdA = sendMePreviousRefrengOrdA;
  }
  if (!sendMeRefreng2)
  {
    sendMeRefrengOrdB = sendMePreviousRefrengOrdB;
  }
  if (sendMeRefrengOrdB == 0 && sendMeRefrengOrdA == 0 && sendMeVersOrdB == 0 &&sendMeVersOrdA == 0 && sendMeIntroOrdA == 0 && sendMeIntroOrdB == 0)
  {
    systemPlayingSound = true; 
  }
  if (sendMeMoodBool == false)
  {
    sendMeMood = sendMePrevMood;
  }
  
  controlChange(0,sendMeMood,0);
  MidiUSB.flush();

  controlChange(1,sendMeIntroOrdA,sendMeIntroOrdB);
  MidiUSB.flush();

  controlChange(2,sendMeVersOrdA,sendMeVersOrdB);
  MidiUSB.flush();

  controlChange(3,sendMeRefrengOrdA,sendMeRefrengOrdB);
  MidiUSB.flush();


}
